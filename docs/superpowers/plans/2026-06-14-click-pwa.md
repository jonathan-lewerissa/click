# CLICK Metronome PWA — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build CLICK — a drift-free, installable setlist metronome PWA — as a single self-contained `index.html` plus `manifest.webmanifest` and `sw.js`, then deploy to Cloudflare Pages.

**Architecture:** Single-page vanilla-JS app. UI state (setlist + current song) is plain JS persisted via a `store` abstraction; the audio engine is a separate imperative look-ahead scheduler driving sample-accurate clicks off the Web Audio clock, with a `requestAnimationFrame` loop reading a scheduled-beat queue to sync visuals. No framework, no bundler, all relative paths.

**Tech Stack:** HTML5, CSS (custom, dark stage theme, Oswald + JetBrains Mono via Google Fonts), vanilla ES, Web Audio API, Service Worker / Cache API, Wake Lock API, `localStorage`.

```js
const CACHE = "click-v1";
const SHELL = [
  "./",
  "./index.html",
  "./manifest.webmanifest",
];

self.addEventListener("install", (e) => {
  e.waitUntil(caches.open(CACHE).then((c) => c.addAll(SHELL)));
  self.skipWaiting();
});

self.addEventListener("activate", (e) => {
  e.waitUntil(
    caches.keys().then((keys) =>
      Promise.all(keys.filter((k) => k !== CACHE).map((k) => caches.delete(k)))
    )
  );
  self.clients.claim();
});

self.addEventListener("fetch", (e) => {
  if (e.request.method !== "GET") return;
  e.respondWith(
    caches.match(e.request).then((hit) => hit || fetch(e.request))
  );
});
```

```js
// --- Storage: window.storage -> localStorage -> in-memory ---
const KEY = "clicklist_v1";
const store = (() => {
  let mem = null;
  const back = (() => {
    try {
      if (window.storage && typeof window.storage.getItem === "function") return window.storage;
      if (window.localStorage) { window.localStorage.getItem(KEY); return window.localStorage; }
    } catch (_) {}
    return null;
  })();
  return {
    get() {
      try {
        const raw = back ? back.getItem(KEY) : mem;
        return raw ? JSON.parse(raw) : null;
      } catch (_) { return null; }
    },
    set(obj) {
      const raw = JSON.stringify(obj);
      try { if (back) back.setItem(KEY, raw); else mem = raw; }
      catch (_) { mem = raw; }
    },
  };
})();

// --- App state ---
const BPM_MIN = 20, BPM_MAX = 300;
const clampBpm = (n) => Math.max(BPM_MIN, Math.min(BPM_MAX, Math.round(n) || BPM_MIN));
const uid = () => "s" + Math.random().toString(36).slice(2, 9);

const state = {
  songs: [],        // [{ id, name, bpm, sig:{beats,unit} }]
  currentId: null,
  playing: false,
};

function load() {
  const saved = store.get();
  if (saved && Array.isArray(saved.songs)) {
    state.songs = saved.songs.map((s) => ({
      id: s.id || uid(),
      name: String(s.name || "Untitled"),
      bpm: clampBpm(s.bpm),
      sig: { beats: (s.sig && s.sig.beats) || 4, unit: (s.sig && s.sig.unit) || 4 },
    }));
    state.currentId = saved.currentId || null;
  }
  if (state.currentId && !state.songs.some((s) => s.id === state.currentId)) {
    state.currentId = null;
  }
}

function save() {
  store.set({ songs: state.songs, currentId: state.currentId });
}

function currentSong() {
  return state.songs.find((s) => s.id === state.currentId) || null;
}
```

```js
  const i = state.songs.findIndex((s) => s.id === id);
  if (i < 0) return;
  state.songs.splice(i, 1);
  if (state.currentId === id) {
    const next = state.songs[i] || state.songs[i - 1] || null;
    selectSong(next ? next.id : null);

  const i = state.songs.findIndex((s) => s.id === id);
  if (i > 0) {
    [state.songs[i - 1], state.songs[i]] = [state.songs[i], state.songs[i - 1]];
    save(); renderAll();
  }
```

- [ ] **Step 1: Add the audio engine to the `<script>` (after state, before render section)**

```js
// --- Audio engine: look-ahead scheduler ---
const audio = {
  ctx: null,
  nextNoteTime: 0,   // audio-clock time of the next click
  beat: 0,           // 0-based beat index within the current bar
  lookahead: 25,     // ms — how often the scheduler runs
  ahead: 0.1,        // s — how far ahead to schedule
  timer: null,
  queue: [],         // [{time, beat}] for the visual loop (Task 5)
};

function ensureCtx() {
  if (!audio.ctx) audio.ctx = new (window.AudioContext || window.webkitAudioContext)();
  if (audio.ctx.state === "suspended") audio.ctx.resume();
  return audio.ctx;
}

function scheduleClick(time, beat) {
  const ctx = audio.ctx;
  const song = currentSong();
  const beats = song ? song.sig.beats : 4;
  const downbeat = beat % beats === 0;

  const osc = ctx.createOscillator();
  const gain = ctx.createGain();
  osc.type = "triangle";
  osc.frequency.value = downbeat ? 1760 : 1108;
  const peak = downbeat ? 0.9 : 0.5;
  gain.gain.setValueAtTime(0.0001, time);
  gain.gain.exponentialRampToValueAtTime(peak, time + 0.002);
  gain.gain.exponentialRampToValueAtTime(0.0001, time + 0.035);
  osc.connect(gain).connect(ctx.destination);
  osc.start(time);
  osc.stop(time + 0.04);

  audio.queue.push({ time, beat: beat % beats });
}

function scheduler() {
  const ctx = audio.ctx;
  const song = currentSong();
  const bpm = song ? song.bpm : 120;
  const beats = song ? song.sig.beats : 4;
  const secondsPerBeat = 60 / bpm;
  while (audio.nextNoteTime < ctx.currentTime + audio.ahead) {
    scheduleClick(audio.nextNoteTime, audio.beat);
    audio.nextNoteTime += secondsPerBeat;
    audio.beat = (audio.beat + 1) % beats;
  }
}

function startClick() {
  const ctx = ensureCtx();
  if (!currentSong()) return;
  audio.beat = 0;
  audio.nextNoteTime = ctx.currentTime + 0.06;
  audio.queue.length = 0;
  state.playing = true;
  audio.timer = setInterval(scheduler, audio.lookahead);
  startVisualLoop(); // defined in Task 5
  acquireWakeLock(); // defined in Task 9
  renderFace();
}

function stopClick() {
  clearInterval(audio.timer);
  audio.timer = null;
  state.playing = false;
  audio.queue.length = 0;
  releaseWakeLock(); // defined in Task 9
  renderFace();
}

function toggleClick() { state.playing ? stopClick() : startClick(); }
```

```js
let rafId = null;
function startVisualLoop() {
  cancelAnimationFrame(rafId);
  const tick = () => {
    if (!state.playing) return;
    const now = audio.ctx.currentTime;
    while (audio.queue.length && audio.queue[0].time <= now) {
      lightBeat(audio.queue.shift().beat);
    }
    rafId = requestAnimationFrame(tick);
  };
  rafId = requestAnimationFrame(tick);
}

function lightBeat(beat) {
  const leds = document.querySelectorAll("#leds .led");
  leds.forEach((l) => l.classList.remove("on"));
  const led = leds[beat];
  if (led) {
    led.classList.add("on");
    setTimeout(() => led.classList.remove("on"), 90);
  }
  const readout = document.querySelector("#face div");
  if (readout) {
    readout.style.transform = "scale(1.06)";
    setTimeout(() => { readout.style.transform = "scale(1)"; }, 80);
  }
}
```

```js
const tapTimes = [];
function tapTempo() {
  const now = performance.now();
  if (tapTimes.length && now - tapTimes[tapTimes.length - 1] > 2000) {
    tapTimes.length = 0; // stale — restart
  }
  tapTimes.push(now);
  if (tapTimes.length > 5) tapTimes.shift();
  if (tapTimes.length >= 2) {
    let sum = 0;
    for (let i = 1; i < tapTimes.length; i++) sum += tapTimes[i] - tapTimes[i - 1];
    const avgMs = sum / (tapTimes.length - 1);
    const bpm = clampBpm(60000 / avgMs);
    const s = currentSong();
    if (s) { s.bpm = bpm; save(); renderFace(); }
  }
}
```

```js
let wakeLock = null;
let wakeWanted = false; // user/playing intent

async function acquireWakeLock() {
  wakeWanted = true;
  if (!("wakeLock" in navigator)) return;
  try {
    wakeLock = await navigator.wakeLock.request("screen");
    wakeLock.addEventListener("release", () => { wakeLock = null; });
  } catch (_) { /* unsupported or denied: silent */ }
}

function releaseWakeLock() {
  wakeWanted = false;
  if (wakeLock) { wakeLock.release().catch(() => {}); wakeLock = null; }
}

// Re-acquire on refocus if still wanted
document.addEventListener("visibilitychange", () => {
  if (document.visibilityState === "visible" && wakeWanted && !wakeLock) {
    acquireWakeLock();
  }
});
```
